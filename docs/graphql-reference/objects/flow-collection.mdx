type FlowCollection {
  any(
    of: FlowCollectionsMeasureable!
  ): String
  block(
    height: BlockSelector
    blockId: HashSelector
  ): FlowBlockDimension!
  count(
    uniq: FlowCollectionUniq
  ): Int
  countBigInt(
    uniq: FlowCollectionUniq
  ): BigInt
  date: Date
  expression(
    get: String!
  ): DecimalNumber
  id: String
  index: Int
  maximum(
    of: FlowCollectionsMeasureable!
    get: FlowCollectionsMeasureable
  ): String
  minimum(
    of: FlowCollectionsMeasureable!
    get: FlowCollectionsMeasureable
  ): String
  signatures: [String!]
  time: DateTime
  transactionsCount: Int
}
