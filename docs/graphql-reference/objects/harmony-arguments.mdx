```graphql
type HarmonyArguments {
  address: String
  any(
    of: HarmonyArgumentsMeasureable!
  ): String
  argIndex: String
  argType: String
  argument(
    argument: String
  ): String
  blockHash(
    blockHash: HashSelector
  ): String
  callPath: String
  count(
    uniq: HarmonyArgumentsUniq
  ): Int
  date: Date
  epoch: BigInt
  expression(
    get: String!
  ): DecimalNumber
  external: Int
  ledger(
    ledger: BlockSelector
  ): BigInt
  maximum(
    of: HarmonyArgumentsMeasureable!
    get: HarmonyArgumentsMeasureable
  ): String
  minimum(
    of: HarmonyArgumentsMeasureable!
    get: HarmonyArgumentsMeasureable
  ): String
  nonce: BigInt
  number: String
  receiver(
    receiver: HashSelector
  ): String
  sender(
    sender: HashSelector
  ): String
  shardId: BigInt
  signatureId: BigInt
  smartContractAddress(
    smartContractAddress: HashSelector
  ): String
  smartContractId(
    smartContractAddress: IntegerSelector
  ): BigInt
  time: DateTime
  toShardId: BigInt
  value: String
}
```
